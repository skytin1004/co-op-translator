# ಕಮಾಂಡ್ ಉಲ್ಲೇಖ

**Co-op Translator** CLI ಅನುವಾದ ಪ್ರಕ್ರಿಯೆಯನ್ನು ಅನೇಕ ಆಯ್ಕೆಗಳ ಮೂಲಕ ಕಸ್ಟಮೈಸ್ ಮಾಡುತ್ತದೆ:

Command                                       | ವಿವರಣೆ
----------------------------------------------|-------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
translate -l "language_codes"                 | ನಿಗದಿತ ಭಾಷೆಗಳಿಗೆ ನಿಮ್ಮ ಪ್ರಾಜೆಕ್ಟ್ ಅನ್ನು ಅನುವಾದಿಸುತ್ತದೆ. ಉದಾಹರಣೆ: translate -l "es fr de" ಸ್ಪ್ಯಾನಿಷ್, ಫ್ರೆಂಚ್ ಮತ್ತು ಜರ್ಮನ್‌ಗೆ ಅನುವಾದಿಸುತ್ತದೆ. translate -l "all" ಬಳಸಿ ಎಲ್ಲಾ ಬೆಂಬಲಿತ ಭಾಷೆಗಳಿಗೆ ಅನುವಾದಿಸಲು.
translate -l "language_codes" -u              | ಇರುವ ಅನುವಾದಗಳನ್ನು ಅಳಿಸಿ ಮರು ಸೃಷ್ಟಿಸುವ ಮೂಲಕ ಅನುವಾದಗಳನ್ನು ನವೀಕರಿಸುತ್ತದೆ. ಎಚ್ಚರಿಕೆ: ಇದು ನಿಗದಿತ ಭಾಷೆಗಳ ಎಲ್ಲಾ ಪ್ರಸ್ತುತ ಅನುವಾದಗಳನ್ನು ಅಳೆಯುತ್ತದೆ.
translate -l "language_codes" -img            | ಕೇವಲ ಚಿತ್ರ ಕಡತಗಳನ್ನು ಅನುವಾದಿಸುತ್ತದೆ.
translate -l "language_codes" -md             | ಕೇವಲ ಮಾರ್ಕ್‌ಡೌನ್ ಕಡತಗಳನ್ನು ಅನುವಾದಿಸುತ್ತದೆ.
translate -l "language_codes" -nb             | ಕೇವಲ ಜ್ಯುಪಿಟರ್ ನೋಟ್‌ಬುಕ್ ಕಡತಗಳನ್ನು (.ipynb) ಅನುವಾದಿಸುತ್ತದೆ.
translate -l "language_codes" --fix           | ಹಿಂದೆ ಮಾಡಲಾದ ಮೌಲ್ಯಮಾಪನ ಫಲಿತಾಂಶಗಳ ಆಧಾರದ ಮೇಲೆ ಕಡಿಮೆ ವಿಶ್ವಾಸಾತ್ಮಕ ಸ್ಕೋರ್ ಹೊಂದಿರುವ ಕಡತಗಳನ್ನು ಮರುಅನುವಾದಿಸುತ್ತದೆ.
translate -l "language_codes" -d              | ವಿವರವಾದ ಲಾಗಿಂಗ್‌ಗಾಗಿ ಡೀಬಗ್ ಮೋಡ್ ಸಕ್ರಿಯಗೊಳಿಸುತ್ತದೆ.
translate -l "language_codes" --save-logs, -s | <root_dir>/logs/ ಫೋಲ್ಡರ್ ಅಡಿಯಲ್ಲಿ DEBUG-ಮಟ್ಟದ ಲಾಗ್‌ಗಳನ್ನು ಫೈಲ್‌ಗಳಿಗೆ ಉಳಿಸುತ್ತದೆ (ಕონსೋಲ್ -d ಮೂಲಕ ನಿಯಂತ್ರಣದಲ್ಲಿರುತ್ತದೆ)
translate -l "language_codes" -r "root_dir"   | ಪ್ರಾಜೆಕ್ಟ್‌ನ ಮೂಲ ಡೈರೆಕ್ಟರಿಯನ್ನು ಸೂಚಿಸುತ್ತದೆ
translate -l "language_codes" -f              | ಚಿತ್ರ ಅನುವಾದಕ್ಕಾಗಿ ಫಾಸ್ಟ್ ಮೋಡ್ ಬಳಸುತ್ತದೆ (ಗುಣಮಟ್ಟ ಮತ್ತು ಹೊಂದಾಣಿಕೆಯಲ್ಲಿ ಸ್ವಲ್ಪ ವ್ಯತ್ಯಾಸದೊಂದಿಗೆ 3x ವೇಗವಾಗಿ ಚಿತ್ರಣ).
translate -l "language_codes" -y              | ಎಲ್ಲಾ ಪ್ರಾಂಪ್ಟ್‌ಗಳನ್ನು ಸ್ವಯಂಚಾಲಿತವಾಗಿ ದೃಢೀಕರಿಸುತ್ತದೆ (CI/CD ಪೈಪ್‌ಲೈನ್ಗಳಿಗಾಗಿ ಉಪಯುಕ್ತ)
translate -l "language_codes" --add-disclaimer/--no-disclaimer | ಅನುವಾದಿತ ಮಾರ್ಕ್‌ಡೌನ್ ಮತ್ತು ನೋಟ್‌ಬುಕ್‌ಗಳಿಗೆ ಯಂತ್ರ ಅನುವಾದ ನಿರಾಕರಣಾ ವಿಭಾಗವನ್ನು ಸೇರಿಸುವ ಅಥವಾ ರದ್ದುಪಡಿಸುವ ಆಯ್ಕೆ (ಪೂರ್ವನಿಯೋಜಿತ: ಸಕ್ರಿಯ).
translate -l "language_codes" --repo-url "https://github.com/org/repo.git" | ನಿಮ್ಮ ರೆಪೊಸಿಟರಿ URL ಬಳಸಿ README ಭಾಷೆಗಳ ವಿಭಾಗ ಸಲಹೆಯನ್ನು ವೈಯಕ್ತೀಕರಿಸುವದು (ಸ್ಪಾರ್ಸ್ ಚೆಕ್‌ಔಟ್).
translate -l "language_codes" --help          | CLI ಒಳಗಿನ ಲಭ್ಯವಿರುವ ಕಮಾಂಡ್‌ಗಳ ಸಹಾಯ ವಿವರಗಳ ಪ್ರದರ್ಶನ
evaluate -l "language_code"                  | ನಿರ್ದಿಷ್ಟ ಭಾಷೆಗಾಗಿ ಅನುವಾದ ಗುಣಮಟ್ಟವನ್ನು ಮೌಲ್ಯಮಾಪನ ಮಾಡಿ ವಿಶ್ವಾಸ ಸ್ಕೋರ್ ನೀಡುತ್ತದೆ
evaluate -l "language_code" -c 0.8           | ಕಸ್ಟಮ್ ವಿಶ್ವಾಸಾತ್ಮಕ ಮಿತಿ ಸಹ ಅಳವಡಿಸಿದ್ದು ಅನುವಾದಗಳನ್ನು ಮೌಲ್ಯಮಾಪನ ಮಾಡುತ್ತದೆ
evaluate -l "language_code" -f               | ವೇಗದ ಮೌಲ್ಯಮಾಪನ ಮೋಡ್ (ನಿಯಮ ಆಧಾರಿತ ಮಾತ್ರ, LLM ಇಲ್ಲದೆ)
evaluate -l "language_code" -D               | ಆಳವಾದ ಮೌಲ್ಯಮಾಪನ ಮೋಡ್ (LLM ಆಧಾರಿತ ಮಾತ್ರ, ಹೆಚ್ಚು ಸಮಗ್ರ ಆದರೆ ನಿಧಾನ)
evaluate -l "language_code" --save-logs, -s  | <root_dir>/logs/ ಅಡಿಯಲ್ಲಿ DEBUG ಮಟ್ಟದ ಲಾಗ್‌ಗಳನ್ನು ಫೈಲ್‌ಗಳಿಗೆ ಉಳಿಸುತ್ತದೆ
migrate-links -l "language_codes"             | ಅನುವಾದಿತ ಮಾರ್ಕ್‌ಡೌನ್ ಕಡತಗಳನ್ನು ಮರುಪ್ರಕ್ರಿಯೆಗೊಳಿಸಿ ನೋಟ್‌ಬುಕ್‌ಗಳ (.ipynb) ಲಿಂಕ್‌ಗಳನ್ನು жаңಿಸುತ್ತದೆ. ಲಭ್ಯವಿದ್ದರೆ ಅನುವಾದಿತ ನೋಟ್‌ಬುಕ್‌ಗಳನ್ನು ಒತ್ತಾಯಿಸುತ್ತದೆ; ಇಲ್ಲದಿದ್ದರೆ ಮೂಲ ನೋಟ್‌ಬುಕ್‌ಗಳಿಗೆ ಹಿಂದಿರುಗಬಹುದು.
migrate-links -l "language_codes" -r          | ಪ್ರಾಜೆಕ್ಟ್ ಮೂಲ ಡೈರೆಕ್ಟರಿಯನ್ನು ಸೂಚಿಸಿ (ಪೂರ್ವನಿಯೋಜಿತ: ಪ್ರಸ್ತುತ ಡೈರೆಕ್ಟರಿ).
migrate-links -l "language_codes" --dry-run   | ಬದಲಾವಣೆಯುಂಟಾಗುವ ಕಡತಗಳನ್ನು ಬರೆದಿಲ್ಲದೆ ತೋರಿಸುವುದು.
migrate-links -l "language_codes" --no-fallback-to-original | ಅನುವಾದಿತ ಬರಹವಿಲ್ಲದಿದ್ದಾಗ ಮೂಲ ನೋಟ್‌ಬುಕ್‌ಗಳಿಗೆ ಲಿಂಕ್‌ಗಳನ್ನು ಮರುಲೇಖನ ಮಾಡಬೇಡಿ (ಅನುವಾದಿತ ಲಿಂಕ್‌ಗಳಿದ್ದಾಗ ಮಾತ್ರ ನವೀಕರಣ).
migrate-links -l "language_codes" -d          | ವಿವರವಾದ ಲಾಗಿಂಗ್‌ಗಾಗಿ ಡೀಬಗ್ ಮೋಡ್ ಸಕ್ರಿಯಗೊಳಿಸುತ್ತದೆ.
migrate-links -l "language_codes" --save-logs, -s | <root_dir>/logs/ ಅಡಿಯಲ್ಲಿ DEBUG-ಮಟ್ಟದ ಲಾಗ್‌ಗಳನ್ನು ಫೈಲ್‌ಗಳಿಗೆ ಉಳಿಸುತ್ತದೆ
migrate-links -l "all" -y                      | ಎಲ್ಲಾ ಭಾಷೆಗಳಿಗೆ ಪ್ರಕ್ರಿಯೆ ಮಾಡಲು ಮತ್ತು ಎಚ್ಚರಿಕೆ ಪ್ರಾಂಪ್ಟ್ ಅನ್ನು ಸ್ವಯಂಚಾಲಿತ ದೃಢೀಕರಣ ನೀಡಲು.

## ಉಪಯೋಗದ ಉದಾಹರಣೆಗಳು

  1. ಪೂರ್ವನಿಯೋಜಿತ ವರ್ತನೆ (ಹಿಂದಿನ ಅನುವಾದಗಳನ್ನು ಅಳಿಸುವದೆನೂ ಇಲ್ಲದೇ ಹೊಸ ಅನುವಾದಗಳನ್ನು ಸೇರಿಸುವುದು):   translate -l "ko"    translate -l "es fr de" -r "./my_project"

  2. ಹೊಸ ಕೋರಿಯನ್ ಚಿತ್ರ ಅನುವಾದಗಳನ್ನು ಮಾತ್ರ ಸೇರಿಸುವುದು (ಹಿಂದಿನ ಅನುವಾದಗಳನ್ನು ಅಳಿಸುವುದಿಲ್ಲ):    translate -l "ko" -img

  3. ಎಲ್ಲಾ ಕೋರಿಯನ್ ಅನುವಾದಗಳನ್ನು ನವೀಕರಿಸುವುದು (ಎಚ್ಚರಿಕೆ: ಮರುಅನುವಾದಿಸುವ ಮುನ್ನ ಎಲ್ಲಾ ಇದ್ದ ಕೋರಿಯನ್ ಅನುವಾದಗಳನ್ನು ಅಳಿಸಲಾಗುತ್ತದೆ):    translate -l "ko" -u

  4. ಕೋರಿಯನ್ ಚಿತ್ರಗಳನ್ನು ಮಾತ್ರ ನವೀಕರಿಸುವುದು (ಎಚ್ಚರಿಕೆ: ಮರುಅನುವಾದಿಸುವ ಮುನ್ನ ಎಲ್ಲಾ ಇದ್ದ ಚಿತ್ರಗಳನ್ನು ಅಳಿಸಲಾಗುತ್ತದೆ):    translate -l "ko" -img -u

  5. ಇತರ ಅನುವಾದಗಳಿಗೆ ತೊಂದರೆಯಿಲ್ಲದೆ ಕೋರಿಯನ್ ಮಾರ್ಕ್‌ಡೌನ್ ಅನುವಾದಗಳನ್ನು ಸೇರಿಸುವುದು:    translate -l "ko" -md

  6. ಹಿಂದಿನ ಮೌಲ್ಯಮಾಪನ ಫಲಿತಾಂಶಗಳ ಆಧಾರದ ಮೇಲೆ ಕಡಿಮೆ ವಿಶ್ವಾಸಾತ್ಮಕ ಅನುವಾದಗಳನ್ನು ಸರಿ ಮಾಡುವುದಕ್ಕಾಗಿ: translate -l "ko" --fix

  7. ನಿರ್ದಿಷ್ಟ ಕಡತಗಳ (ಮಾರ್ಕ್‌ಡೌನ್) ಮಾತ್ರ ಕಡಿಮೆ ವಿಶ್ವಾಸಾತ್ಮಕ ಅನುವಾದಗಳನ್ನು ಸರಿಪಡಿಸುವುದು: translate -l "ko" --fix -md

  8. ನಿರ್ದಿಷ್ಟ ಕಡತಗಳ (ಚಿತ್ರ) ಮಾತ್ರ ಕಡಿಮೆ ವಿಶ್ವಾಸಾತ್ಮಕ ಅನುವಾದಗಳನ್ನು ಸರಿಪಡಿಸುವುದು: translate -l "ko" --fix -img

  9. ಚಿತ್ರ ಅನುವಾದಕ್ಕೆ ಫಾಸ್ಟ್ ಮೋಡ್ ಬಳಸಿ:    translate -l "ko" -img -f

  10. ಕಸ್ಟಮ್ ಪದವಿರಾಮದೊಂದಿಗೆ ಕಡಿಮೆ ವಿಶ್ವಾಸಾತ್ಮಕ ಅನುವಾದಗಳನ್ನು ಸರಿಪಡಿಸುವುದು: translate -l "ko" --fix -c 0.8

  11. ಡೀಬಗ್ ಮೋಡ್ ಉದಾಹರಣೆ: - translate -l "ko" -d: ಡೀಬಗ್ ಲಾಗಿಂಗ್ ಸಕ್ರಿಯಗೊಳಿಸಿ.
  12. ಲಾಗ್‌ಗಳನ್ನು ಫೈಲ್‌ಗಳಿಗೆ ಉಳಿಸಿ: translate -l "ko" -s
  13. ಕન્સೋಲ್ DEBUG ಮತ್ತು ಫೈಲ್ DEBUG: translate -l "ko" -d -s
  14. ಯಂತ್ರ ಅನುವಾದ ನಿರಾಕರಣಾ ವಿಭಾಗವನ್ನು ಸೇರಿಸದೆ ಅನುವಾದಿಸಿ: translate -l "ko" --no-disclaimer

  15. ಕೋರಿಯನ್ ಅನುವಾದಗಳಿಗಾಗಿ ನೋಟ್ಭುಕ್ ಲಿಂಕ್‌ಗಳನ್ನು ಮೈಗ್ರೇಟ್ ಮಾಡಿ (ಲಭ್ಯವಿದ್ದರೆ ಅನುವಾದಿತ ನೋಟ್ಭುಕ್‌ಗಳಿಗೆ ಲಿಂಕ್‌ಗಳನ್ನು ನವೀಕರಿಸಿ):    migrate-links -l "ko"

  15. ಡ್ರೈ-ರನ್ ಸಹಿತ ಲಿಂಕ್‌ಗಳನ್ನು ಮೈಗ್ರೇಟ್ ಮಾಡಿ (ಯಾವುದೇ ಕಡತ ಬರೆದಿಲ್ಲ):    migrate-links -l "ko" --dry-run

  16. ಅನುವಾದಿತ ನೋಟ್ಭುಕ್‌ಗಳು ಇದ್ದಾಗ ಮಾತ್ರ ಲಿಂಕ್ ನವೀಕರಣ ಮಾಡು (ಮೂಲಗಳ ಕಡೆಗೆ ಹಿಂದಿರುಗಬೇಡಿ):    migrate-links -l "ko" --no-fallback-to-original

  17. ದೃಢೀಕರಣ ಪ್ರಾಂಪ್ಟ್ తో ಎಲ್ಲಾ ಭಾಷೆಗಳನ್ನು ಪ್ರಕ್ರಿಯೆ ಮಾಡು:    migrate-links -l "all"

  18. ಎಲ್ಲಾ ಭಾಷೆಗಳನ್ನು ಪ್ರಕ್ರಿಯೆ ಮಾಡಿ ಸ್ವಯಂಚಾಲಿತ ದೃಢೀಕರಣ ನೀಡಿ:    migrate-links -l "all" -y
  19. migrate-links ಗಾಗಿ ಲಾಗ್‌ಗಳನ್ನು ಫೈಲ್‌ಗಳಿಗೆ ಉಳಿಸಿ:    migrate-links -l "ko ja" -s

  20. ನಿಮ್ಮ ರೆಪೊ URL ಬಳಸಿ README ಭಾಷೆಗಳ ಸಲಹೆಯನ್ನು ವೈಯಕ್ತೀಕರಿಸಿ:
      translate -l "ko" --repo-url "https://github.com/org/repo.git"

### ಮೌಲ್ಯಮಾಪನ ಉದಾಹರಣೆಗಳು

> [!WARNING]  
> **ಬೀಟಾ ವೈಶಿಷ್ಟ್ಯ**: ಮೌಲ್ಯಮಾಪನ ಕಾರ್ಯಾಚರಣೆ ಈಗಾಗಲೇ ಬೀಟಾ ಸ್ಥಿತಿಯಲ್ಲಿ ಇದೆ. ಈ ವೈಶಿಷ್ಟ್ಯ ಅನುವಾದಿತ ಡಾಕ್ಯುಮೆಂಟ್‌ಗಳನ್ನು ಮೌಲ್ಯಮಾಪನ ಮಾಡಲು ಬಿಡುಗಡೆಯಾಗಿದೆ, ಮತ್ತು ಮೌಲ್ಯಮಾಪನ ವಿಧಾನ ಮತ್ತು ವಿವರವಾದ ಅನುಷ್ಠಾನ ಇನ್ನೂ ಅಭಿವೃದ್ಧಿಯಲ್ಲಿದ್ದು ಬದಲಾವಣೆಗಳಿಗೆ ಒಳಪಟ್ಟಿದೆ.

  1. ಕೋರಿಯನ್ ಅನುವಾದಗಳನ್ನು ಮೌಲ್ಯಮಾಪನ ಮಾಡಿ: evaluate -l "ko"

  2. ಕಸ್ಟಮ್ ವಿಶ್ವಾಸಾತ್ಮಕ ಮಿತಿ ನೆರವಾಗಿ ಮೌಲ್ಯಮಾಪನ: evaluate -l "ko" -c 0.8

  3. ವೇಗದ ಮೌಲ್ಯಮಾಪನ (ನಿಯಮ ಆಧಾರಿತ ಮಾತ್ರ): evaluate -l "ko" -f

  4. ಆಳವಾದ ಮೌಲ್ಯಮಾಪನ (LLM ಆಧಾರಿತ ಮಾತ್ರ): evaluate -l "ko" -D
